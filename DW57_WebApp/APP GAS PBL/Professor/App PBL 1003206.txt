function doGet() {
  // Renderiza o arquivo HTML
  return HtmlService.createHtmlOutputFromFile('Index')
    .setTitle('Meu App Vinculado')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

function salvarDados(obj) {
  // Como o script está vinculado, usamos 'getActiveSpreadsheet'
  var planilha = SpreadsheetApp.getActiveSpreadsheet();
  var aba = planilha.getSheets()[0]; // Pega a primeira aba

  // Regra de Ouro: Lógica de Status
  var status = (obj.urgencia === "Alta") ? "IMEDIATO" : "Pendente";
  
  // Adiciona a linha com os dados recebidos do HTML
  aba.appendRow([
    new Date(), 
    obj.nome, 
    obj.email.toLowerCase(),
    obj.telefone,
    status,
    obj.urgencia,
    obj.mensagem
  ]);
  
  return "Feedback registrado com sucesso";
}

function obterTotalCadastros() {
  var planilha = SpreadsheetApp.getActiveSpreadsheet();
  let aba = planilha.getSheets()[0];

  let registros = aba.getLastRow() - 1;

  return registros > 0 ? registros: 0;
}


// Desafio Extra
function buscarUltimoNome() {
  var planilha = SpreadsheetApp.getActiveSpreadsheet();
  let aba = planilha.getSheets()[0];

  let nome = aba.getRange(aba.getLastRow(), 2).getValue();

  return nome;
}

    function ultimoCadastrado() {
      let ultimoNome = document.getElementById("ultimoNome");

      ultimoNome.innerText = "Buscando...";

      google.script.run.withSuccessHandler(
        function(nome) {
          ultimoNome.innerText = nome;
        }
      ).buscarUltimoNome();
    }

    function atualizarContador() {
      let contador = document.getElementById("contador");

      contador.innerText = "Buscando...";

      google.script.run.withSuccessHandler(
        function(valor) {
          contador.innerText = valor;
        }
      ).obterTotalCadastros();
    }

    function enviar() {
      // 1. Coleta os valores dos campos
      let cmp_nome = document.getElementById('nome');
      let cmp_email = document.getElementById('email');
      let cmp_telefone = document.getElementById('telefone');
      let cmp_urgencia = document.getElementById('urgencia');
      let cmp_mensagem = document.getElementById('mensagem');

      // --- INÍCIO DA VALIDAÇÃO (DESAFIO 1) ---
      // Verifica se o nome OU o email estão vazios
      if (cmp_nome.value  === "" || cmp_email.value === "") {
        alert("Por favor, preencha todos os campos antes de salvar!");
        return; // O 'return' para a execução aqui e não envia para o Google Sheets
      }
      // --- FIM DA VALIDAÇÃO ---

      // Capturamos o botão para podermos alterá-lo
      var btn = document.querySelector("button"); 

      var dados = {
        nome: cmp_nome.value,
        email: cmp_email.value,
        telefone: cmp_telefone.value,
        urgencia: cmp_urgencia.value,
        mensagem: cmp_mensagem.value
      };

      // --- INÍCIO DO FEEDBACK (DESAFIO 2) ---
      // Desabilitamos o botão e mudamos o texto
      btn.disabled = true;
      btn.innerText = "Enviando para a Planilha...";
      btn.style.backgroundColor = "#ccc"; // Opcional: muda a cor para cinza
      // --- FIM DO FEEDBACK ---

      // Chama a função salvarDados do arquivo Código.gs
      google.script.run
        .withSuccessHandler(function(res) {
          alert(res);

          cmp_nome.value = "";
          cmp_email.value = "";
          cmp_telefone.value = "";
          cmp_urgencia.value = "";
          cmp_mensagem.value = "";

        // --- RESTAURANDO O BOTÃO ---
        // Após o sucesso, o botão precisa voltar a funcionar
        btn.disabled = false;
        btn.innerText = "Salvar na Planilha";
        btn.style.backgroundColor = "#4285f4"; // Volta para a cor original
        })
        .salvarDados(dados);
    }
  </script>
